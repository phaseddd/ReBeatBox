    ```java
    public class PadButton extends JButton {
        private int midiNote;
        private String padLabel;
        private final String defaultLabel;
        private final int defaultMidiNote;
        private final RealtimeReceiver receiver;
        private final Timer holdTimer;
        private boolean pressedVisual = false;

        /** MIDI Channel 10 for GM percussion (0-indexed: 9). */
        private static final int DRUM_CHANNEL = 10;

        /** Fixed velocity per D-02/D-05. */
        private static final int VELOCITY = 100;

        public PadButton(String label, int midiNote, RealtimeReceiver receiver) {
            super(label);
            this.padLabel = label;
            this.midiNote = midiNote;
            this.defaultLabel = label;
            this.defaultMidiNote = midiNote;
            this.receiver = receiver;

            // Non-focusable — drum pads don't participate in keyboard focus
            setFocusable(false);

            // Square shape per UI-SPEC
            setPreferredSize(new Dimension(48, 48));

            // Font for pad label
            setFont(new Font("SansSerif", Font.PLAIN, 10));
            setForeground(new Color(0xCCCCCC)); // default text color

            // Custom background and border (Radiance theme may override — use setBackground)
            setBackground(new Color(0x1A1A2E));
            setBorder(BorderFactory.createLineBorder(new Color(0x2A3A5E), 1));

            // 200ms one-shot timer for pressed visual hold
            holdTimer = new Timer(200, e -> {
                pressedVisual = false;
                repaint();
            });
            holdTimer.setRepeats(false);

            setupMouseListener();
        }

        private void setupMouseListener() {
            addMouseListener(new MouseAdapter() {
                @Override
                public void mousePressed(MouseEvent e) {
                    if (SwingUtilities.isLeftMouseButton(e)) {
                        sendNoteOn();
                        pressedVisual = true;
                        repaint();
                    }
                }

                @Override
                public void mouseReleased(MouseEvent e) {
                    if (SwingUtilities.isLeftMouseButton(e)) {
                        sendNoteOff();
                        holdTimer.restart(); // 200ms then restore visual
                    }
                }

                @Override
                public void mouseEntered(MouseEvent e) {
                    if (!pressedVisual) {
                        repaint();
                    }
                }

                @Override
                public void mouseExited(MouseEvent e) {
                    if (!pressedVisual) {
                        repaint();
                    }
                }

                @Override
                public void mouseClicked(MouseEvent e) {
                    if (SwingUtilities.isRightMouseButton(e)) {
                        showAssignMenu(e.getComponent(), e.getX(), e.getY());
                    }
                }
            });
        }
    ```

    ```java
        @Override
        protected void paintComponent(Graphics g) {
            Graphics2D g2d = (Graphics2D) g;

            ButtonModel model = getModel();
            boolean hover = model.isRollover();
            boolean pressed = pressedVisual || model.isPressed();

            // Select fill color per state
            Color fillColor;
            if (pressed) {
                fillColor = new Color(0x003344); // pressed — dark cyan
            } else if (hover) {
                fillColor = new Color(0x25304A); // hover — lighter navy
            } else {
                fillColor = new Color(0x1A1A2E); // default
            }
            g2d.setColor(fillColor);
            g2d.fillRect(0, 0, getWidth(), getHeight());

            // Select border color per state
            Color borderColor;
            int borderWidth;
            if (pressed) {
                borderColor = new Color(0x00E5FF); // neon cyan, 2px
                borderWidth = 2;
            } else if (hover) {
                borderColor = new Color(0xE040FB); // neon magenta, 1px
                borderWidth = 1;
            } else {
                borderColor = new Color(0x2A3A5E); // default, 1px
                borderWidth = 1;
            }
            g2d.setColor(borderColor);
            g2d.setStroke(new BasicStroke(borderWidth));
            g2d.drawRect(borderWidth / 2, borderWidth / 2,
                getWidth() - borderWidth, getHeight() - borderWidth);

            // Text color
            if (pressed) {
                setForeground(new Color(0x00E5FF));
            } else if (hover) {
                setForeground(new Color(0xE0E0E0));
            } else {
                setForeground(new Color(0xCCCCCC));
            }

            // Draw text via super (handles alignment and font)
            super.paintComponent(g);
        }
    ```
