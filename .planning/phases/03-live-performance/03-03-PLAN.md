    ```java
    import javax.swing.*;
    import java.awt.*;

    /**
     * Three-row virtual piano keyboard displaying the D-01 keyboard-to-MIDI mapping.
     * Renders 36+ keys across number row (C5-C6), QWERTY row (C4-C5), and bottom row (C3-C4).
     * Keys highlight in neon cyan when pressed via {@link #setKeyHighlighted(int, boolean)}.
     *
     * <p>Uses custom paintComponent — no JButton subcomponents per Pattern 3 (RESEARCH.md).
     */
    public class KeyboardHintPanel extends JPanel {

        // --- Layout constants ---
        private static final int ROW_COUNT = 3;
        private static final int ROW_HEIGHT = 28;   // per UI-SPEC
        private static final int PANEL_HEIGHT = ROW_COUNT * ROW_HEIGHT; // 84px

        // Black key proportions (UI-SPEC)
        private static final double BLACK_KEY_WIDTH_RATIO = 0.60;
        private static final double BLACK_KEY_HEIGHT_RATIO = 0.70;

        // --- Color constants (UI-SPEC §Color contract) ---
        private static final Color BG_COLOR = new Color(0x0a0a14);

        // White key colors
        private static final Color WHITE_IDLE_FILL = new Color(0x2A2A2A);
        private static final Color WHITE_IDLE_BORDER = new Color(0x3A3A3A);
        private static final Color WHITE_IDLE_LABEL = new Color(0x808080);

        // Black key colors
        private static final Color BLACK_IDLE_FILL = new Color(0x1A1A1A);
        private static final Color BLACK_IDLE_BORDER = new Color(0x2A2A2A);
        private static final Color BLACK_IDLE_LABEL = new Color(0x606060);

        // Pressed highlight (neon cyan per D-08)
        private static final Color PRESSED_BORDER = new Color(0x00E5FF);
        private static final Color PRESSED_LABEL = new Color(0x00E5FF);
        // Pressed fill: neon cyan at reduced alpha over idle fill
        private static final Color WHITE_PRESSED_FILL = new Color(0x00E5FF, 64);  // alpha ~0.25
        private static final Color BLACK_PRESSED_FILL = new Color(0x00E5FF, 51);  // alpha ~0.20

        // --- Font ---
        private static final Font KEY_LABEL_FONT = new Font("SansSerif", Font.PLAIN, 11);
        private static final Font NOTE_LABEL_FONT = new Font("Monospaced", Font.PLAIN, 9);

        // --- Highlight state: keyCode -> isHighlighted ---
        private final Map<Integer, Boolean> keyHighlights = new HashMap<>();

        /**
         * Creates the keyboard hint panel with dark background and double-buffering.
         */
        public KeyboardHintPanel() {
            setBackground(BG_COLOR);
            setDoubleBuffered(true);
            setPreferredSize(new Dimension(800, PANEL_HEIGHT));
        }

        /**
         * Sets the highlight state for a key identified by its AWT keyCode.
         * Triggers an immediate repaint of the affected area.
         *
         * @param keyCode      the AWT KeyEvent key code (e.g., KeyEvent.VK_Q)
         * @param highlighted  true to show pressed neon cyan state, false for idle
         */
        public void setKeyHighlighted(int keyCode, boolean highlighted) {
            boolean changed = highlighted != keyHighlights.getOrDefault(keyCode, false);
            keyHighlights.put(keyCode, highlighted);
            if (changed) {
                repaint(); // repaint entire panel — custom paint is fast
            }
        }

        /**
         * Returns whether a key is currently highlighted.
         */
        public boolean isKeyHighlighted(int keyCode) {
            return keyHighlights.getOrDefault(keyCode, false);
        }

        /**
         * Clears all key highlights. Called on window focus loss to prevent
         * stuck highlights when the user Alt+Tabs away.
         */
        public void clearAllHighlights() {
            if (!keyHighlights.isEmpty()) {
                keyHighlights.clear();
                repaint();
            }
        }

        // --- Painting (Task 2 implements the body) ---

        @Override
        protected void paintComponent(Graphics g) {
            super.paintComponent(g);
            Graphics2D g2d = (Graphics2D) g;

            int w = getWidth();
            int h = getHeight();
            if (w <= 0 || h <= 0) return;

            // Enable antialiasing for text and lines
            g2d.setRenderingHint(RenderingHints.KEY_ANTIALIASING, RenderingHints.VALUE_ANTIALIAS_ON);
            g2d.setRenderingHint(RenderingHints.KEY_TEXT_ANTIALIASING, RenderingHints.VALUE_TEXT_ANTIALIAS_ON);

            // Background fill
            g2d.setColor(BG_COLOR);
            g2d.fillRect(0, 0, w, h);

            // Draw three rows
            drawKeyRow(g2d, 0); // Number row (top)
            drawKeyRow(g2d, 1); // QWERTY row (middle)
            drawKeyRow(g2d, 2); // Bottom row
        }
    }
    ```

    **drawKeyRow method:**

    ```java
    private void drawKeyRow(Graphics2D g2d, int rowIndex) {
        KeyDef[] keys = ALL_ROWS[rowIndex];
        if (keys.length == 0) return;

        int w = getWidth();
        int rowY = rowIndex * ROW_HEIGHT;

        // Calculate white key width from available width and count of white keys
        int whiteKeyCount = 0;
        for (KeyDef key : keys) {
            if (!key.isBlack) whiteKeyCount++;
        }
        if (whiteKeyCount == 0) return;
        int whiteKeyWidth = w / whiteKeyCount;

        // --- Pass 1: Draw white keys first ---
        int whiteIndex = 0;
        for (KeyDef key : keys) {
            if (key.isBlack) continue;
            int keyX = whiteIndex * whiteKeyWidth;
            int keyW = whiteKeyWidth - 1; // 1px gap

            drawSingleKey(g2d, key, keyX, rowY, keyW, ROW_HEIGHT);
            whiteIndex++;
        }

        // --- Pass 2: Draw black keys on top ---
        int blackKeyWidth = (int) (whiteKeyWidth * BLACK_KEY_WIDTH_RATIO);
        int blackKeyHeight = (int) (ROW_HEIGHT * BLACK_KEY_HEIGHT_RATIO);
        int blackKeyYOffset = (ROW_HEIGHT - blackKeyHeight) / 2;

        whiteIndex = 0;
        for (int i = 0; i < keys.length; i++) {
            KeyDef key = keys[i];
            if (whiteIndex < whiteKeyCount && keys[i].isBlack) {
                // Black key is centered between the previous white key and next white key
                // Find the previous white key's index
                int prevWhiteIdx = i;
                while (prevWhiteIdx >= 0 && keys[prevWhiteIdx].isBlack) prevWhiteIdx--;
                int prevWhiteX = prevWhiteIdx >= 0 ? getWhiteKeyX(keys, prevWhiteIdx, whiteKeyWidth) : 0;

                // Find next white key's index (or use end of row)
                int nextWhiteIdx = i + 1;
                while (nextWhiteIdx < keys.length && keys[nextWhiteIdx].isBlack) nextWhiteIdx++;
                int nextWhiteX = nextWhiteIdx < keys.length ? getWhiteKeyX(keys, nextWhiteIdx, whiteKeyWidth) : w;

                // Center black key between adjacent white keys
                int midX = (prevWhiteX + whiteKeyWidth + nextWhiteX) / 2;
                int blackX = midX - blackKeyWidth / 2;

                drawSingleKey(g2d, key, blackX, rowY + blackKeyYOffset, blackKeyWidth, blackKeyHeight);
            }
            if (!keys[i].isBlack) whiteIndex++;
        }
    }
    ```

    **drawSingleKey method (per-state color contract):**

    ```java
    private void drawSingleKey(Graphics2D g2d, KeyDef key, int x, int y, int width, int height) {
        boolean pressed = isKeyHighlighted(key.keyCode);

        // Select colors based on key type and state (UI-SPEC color contract)
        Color fillColor, borderColor, labelColor;
        int borderWidth;

        if (key.isBlack) {
            if (pressed) {
                fillColor = BLACK_PRESSED_FILL;
                borderColor = PRESSED_BORDER;
                labelColor = PRESSED_LABEL;
                borderWidth = 2;
            } else {
                fillColor = BLACK_IDLE_FILL;
                borderColor = BLACK_IDLE_BORDER;
                labelColor = BLACK_IDLE_LABEL;
                borderWidth = 1;
            }
        } else {
            if (pressed) {
                fillColor = WHITE_PRESSED_FILL;
                borderColor = PRESSED_BORDER;
                labelColor = PRESSED_LABEL;
                borderWidth = 2;
            } else {
                fillColor = WHITE_IDLE_FILL;
                borderColor = WHITE_IDLE_BORDER;
                labelColor = WHITE_IDLE_LABEL;
                borderWidth = 1;
            }
        }

        // Fill
        g2d.setColor(fillColor);
        g2d.fillRect(x, y, width, height);

        // Border
        g2d.setColor(borderColor);
        g2d.setStroke(new BasicStroke(borderWidth));
        g2d.drawRect(x, y, width, height);

        // Label — key name (e.g., "Q") at top
        if (height >= 18 && width >= 12) {
            g2d.setFont(KEY_LABEL_FONT);
            g2d.setColor(labelColor);
            FontMetrics fm = g2d.getFontMetrics();
            String label = key.keyLabel;
            int labelX = x + (width - fm.stringWidth(label)) / 2;
            int labelY = y + fm.getAscent() + 1;
            g2d.drawString(label, labelX, labelY);

            // Note name (e.g., "C4") at bottom — smaller, muted
            if (height >= 28 && !key.isBlack) { // only on white keys which have more space
                g2d.setFont(NOTE_LABEL_FONT);
                g2d.setColor(pressed ? PRESSED_LABEL : WHITE_IDLE_LABEL);
                FontMetrics nfm = g2d.getFontMetrics();
                String noteStr = key.noteName;
                int noteX = x + (width - nfm.stringWidth(noteStr)) / 2;
                int noteY = y + height - 3;
                g2d.drawString(noteStr, noteX, noteY);
            }
        }
    }
    ```
